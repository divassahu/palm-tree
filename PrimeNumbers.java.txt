public class PrimeNumbers {
    public static void main(String[] args) {
        System.out.println("Prime numbers up to 30 are: ");
        for (int i = 2; i <= 30; i++) {
            boolean isPrime = true;
            for (int j = 2; j < i; j++) {
                if (i % j == 0) {
                    isPrime = false;
                    break;
                }
            }
            if (isPrime) {
                System.out.print(i + " ");
            }
        }
    }
}
